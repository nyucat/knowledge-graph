from ltp import LTP

# 加载预训练中文NER模型
ltp = LTP("LTP/small")

# 待处理文本
text = "艾伦·麦席森·图灵，英国数学家，1931年进入剑桥大学国王学院，导师是阿隆佐·丘奇。"

文本中同时出现的实体：图灵、布莱切利园、恩尼格玛密码、二战、阿隆佐·丘奇

消歧结果：「图灵」必然指向「艾伦・麦席森・图灵」，而非其他同名实体。
